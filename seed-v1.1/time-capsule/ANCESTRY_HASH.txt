Companion Script — Generate Real Hashes (bundle script)

generate_ancestry_hash.sh

#!/bin/bash

OUTPUT="ANCESTRY_HASH.txt"
TMP="ANCESTRY_HASH_TMP.txt"

echo "Generating ORI-GARDEN Seed v1.1 Ancestry Hashes..."

FILES=(
GENESIS.md
LINEAGE.md
ARCHITECTURE.md
CIVILIZATION_STACK.json
SYMBIOSIS_PROTOCOL_SPEC.json
ORIGIN_PARADOX.md
EPOCH_TIMESTAMP.json
CIVILIZATION_STACK_ARCHAEOLOGY.md
CIVILIZATION_STACK_TIME_AXIS.json
CIVILIZATION_STACK_SIMULATION.md
ORIGIN_PARADOX_EXTENDED.md
SYMBIOSIS_FLOW_DIAGRAM.md
RUNTIME_FLOW_DIAGRAM.svg
IMPLEMENTATION_GUIDE.md
DECODER_MANIFEST.txt
EASTER_EGG_DECODER_HINT.txt
MYTH_LAYER.txt
TIME_CAPSULE_BUNDLE.sig
)

> $TMP
for f in "${FILES[@]}"; do
  if [ -f "$f" ]; then
    HASH=$(sha256sum "$f" | awk '{print $1}')
    echo "$f" >> $TMP
    echo "SHA256: $HASH" >> $TMP
    echo "" >> $TMP
  fi
done

ROOT=$(sha256sum $TMP | awk '{print $1}')
echo "ANCESTRY_ROOT_SHA256: $ROOT" >> $TMP

mv $TMP $OUTPUT
echo "Done. Root hash: $ROOT"
